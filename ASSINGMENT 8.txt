DECLARE str1, str2 : STRING
DECLARE char1, char2 : CHAR
DECLARE i : INTEGER


str1 ← ""
str2 ← ""
char1 ← ''	
char2 ← ''
i ← 0

INPUT "Please enter string to be processed: ", str1

str1 ← LOWER(str1)
char1 ← LEFT(str1, 1)

str2 ← str2 + char1
FOR i ← 1 TO LEN(str1)
    char2 ← MID(str1, i + 1, 1)
    IF char2 ← char1 THEN
       char2 ← "*"

    ENDIF

    str2 ← str2 + char2

NEXT i

OUTPUT "Output STRING is: ", str2
