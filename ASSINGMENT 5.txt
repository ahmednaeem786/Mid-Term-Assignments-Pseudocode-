// ASSINGMENT 5
//Finding Vowels In A STRING

DECLARE str1 : STRING
DECLARE i, vowelscounter : INTEGER
DECLARE char1 : CHAR
DECLARE acounter, ecounter, icounter, ocounter, ucounter : INTEGER


str1 ← ""
char1 ← ""
i ← 0
vowelscounter ← 0
acounter ← 0
ecounter ← 0
icounter ← 0
ocounter ← 0
ucounter ← 0


INPUT "Please Enter STRING Which Is To Be Processed: ", str1

FOR i ← 1 TO LEN(str1)
    char1 ← MID(str1, i, 1)
    char1 ← UPPER(char1)
    IF char1 ← "A" THEN

vowelscounter ← vowelscounter + 1
acounter ← acounter + 1

    ELSEIF char1 ← "E" THEN
ecounter ← ecounter + 1
vowelscounter ← vowelscounter + 1

    ELSEIF char1 ← "I" THEN
icounter ← icounter + 1
vowelscounter ← vowelscounter + 1
      
    ELSEIF char1 ← "O" THEN
ocounter ← ocounter + 1
vowelscounter ← vowelscounter + 1
    
    ElSEIF char1 ← "U" THEN
ucounter ← ucounter + 1
vowelscounter ← vowelscounter + 1
    ENDIF


NEXT i

OUTPUT "Vowels In STRING Are: ", vowelscounter
OUTPUT "Number Of A's are: ", acounter
OUTPUT "Number Of E's are: ", ecounter
OUTPUT "Number Of I's are: ", icounter
OUTPUT "Number Of O's are: ", ocounter
OUTPUT "Number Of U's are: ", ucounter
