//ASSINGMENT 4
//Find the character that appears most number of times in an
// entered string and output it.

DECLARE str1 : STRING
DECLARE char1 : CHAR
DECLARE char2 : CHAR
DECLARE char3 : CHAR
DECLARE highest : INTEGER
DECLARE countoccur : INTEGER
DECLARE i : INTEGER
DECLARE j : INTEGER


str1 ← ""
char1 ← ""
i ← 0
char2 ← ""
char3 ← ""
highest ← 0
j ← 0


INPUT "Please Enter STRING: ", str1

FOR i ← 1 TO LEN(str1)
    char1 ← MID(str1, i, 1)
    countoccur ← 0
NEXT i

FOR j ← 1 TO LEN(str1)
    char2 ← MID(str1, j, 1)
    IF char2 ← char1 THEN countoccur ← countoccur + 1
NEXT j

IF countoccur > highest THEN
    highest ← countoccur
    char3 ← char1
ENDIF


OUTPUT "This character Appeared: ", char1
OOUTPUT countoccur, " number of times"
